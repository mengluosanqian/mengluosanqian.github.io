git常用命令 

git add .  提交所有文件
git checkout xxx   还原到以前的状态
git commit -m "xxx"
git push origin master   推送到远程分支
git pull origin master  从远处分支拉取代码
git branch 查看所有分支
git checkout -b xxx 创建一个新的分支
git merge xxx  合并xxx分支到当前分支
git clone 克隆
git status 查看git状态
git config user.name lw
git config user.email lw@...
git stash 暂时搁置修改的东西
git stash pop 取出暂存的代码
git fetch 将远程主机的最新内容拉到本地
git reset --hard origin/master 覆盖
